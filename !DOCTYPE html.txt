    <script>
        // Initialize Icons
        lucide.createIcons();

        // Admin System
        const ADMIN_PASS = "$@M!2213590786";
        let isAdmin = false;

        const loginBtn = document.getElementById('loginBtn');
        const logoutBtn = document.getElementById('logoutBtn');
        const loginModal = document.getElementById('loginModal');
        const adminPassInput = document.getElementById('adminPass');
        const verifyLogin = document.getElementById('verifyLogin');
        const closeLogin = document.getElementById('closeLogin');

        loginBtn.addEventListener('click', () => {
            loginModal.classList.remove('hidden');
            loginModal.classList.add('flex');
        });

        closeLogin.addEventListener('click', () => {
            loginModal.classList.add('hidden');
            loginModal.classList.remove('flex');
        });

        verifyLogin.addEventListener('click', () => {
            if (adminPassInput.value === ADMIN_PASS) {
                activateAdmin();
                loginModal.classList.add('hidden');
                loginModal.classList.remove('flex');
                adminPassInput.value = '';
                alert("Admin Mode Activated.");
            } else {
                alert("Incorrect Credentials.");
                adminPassInput.value = '';
            }
        });

        function activateAdmin() {
            isAdmin = true;
            document.body.classList.add('is-admin');
            loginBtn.classList.add('hidden');
            logoutBtn.classList.remove('hidden');
            localStorage.setItem('smrs_admin', 'true');
        }

        logoutBtn.addEventListener('click', () => {
            isAdmin = false;
            document.body.classList.remove('is-admin');
            loginBtn.classList.remove('hidden');
            logoutBtn.classList.add('hidden');
            localStorage.removeItem('smrs_admin');
            window.location.reload();
        });

        // Editing Functionality
        let currentEditingId = null;
        function editContent(id) {
            if (!isAdmin) return;
            currentEditingId = id;
            const element = document.getElementById(id);
            const editTextarea = document.getElementById('editTextarea');
            editTextarea.value = element.innerText;
            document.getElementById('editModal').classList.remove('hidden');
        }

        document.getElementById('saveEditBtn').addEventListener('click', () => {
            const newValue = document.getElementById('editTextarea').value;
            if (currentEditingId) {
                document.getElementById(currentEditingId).innerText = newValue;
                // Save to localStorage to persist for current user session/browser
                localStorage.setItem(`content_${currentEditingId}`, newValue);
            }
            closeModal('editModal');
        });

        function closeModal(modalId) {
            document.getElementById(modalId).classList.add('hidden');
        }

        // Image Upload Simulation
        function setupImageUpload(inputId, targetImgId) {
            document.getElementById(inputId).addEventListener('change', function(e) {
                if (e.target.files && e.target.files[0]) {
                    const reader = new FileReader();
                    reader.onload = function(event) {
                        document.getElementById(targetImgId).src = event.target.result;
                        localStorage.setItem(`img_${targetImgId}`, event.target.result);
                    }
                    reader.readAsDataURL(e.target.files[0]);
                }
            });
        }

        setupImageUpload('upload-hero-img', 'hero-img');
        setupImageUpload('upload-writer-img', 'writer-img');

        // Load Persisted Content
        window.addEventListener('load', () => {
            const editableElements = ['rebel-vision', 'rebel-title', 'writer-title', 'photo-title', 'hero-name'];
            editableElements.forEach(id => {
                const el = document.getElementById(id);
                if(el) {
                    const saved = localStorage.getItem(`content_${id}`);
                    if (saved) el.innerText = saved;
                    
                    // Add click listeners if admin
                    el.addEventListener('click', () => {
                        if (isAdmin) editContent(id);
                    });
                    el.title = "Click to edit (Admin)";
                }
            });

            const imgs = ['hero-img', 'writer-img'];
            imgs.forEach(id => {
                const savedImg = localStorage.getItem(`img_${id}`);
                if (savedImg) document.getElementById(id).src = savedImg;
            });
        });

        // Intersection Observer for scroll animations
        const observerOptions = { threshold: 0.1 };
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('fade-in');
                    observer.unobserve(entry.target);
                }
            });
        }, observerOptions);

        document.querySelectorAll('section').forEach(section => {
            section.style.opacity = "0";
            observer.observe(section);
        });

        // Add explicit CSS class for fade-in effect to handle initial state
        const style = document.createElement('style');
        style.innerHTML = `
            section.fade-in {
                opacity: 1 !important;
                transition: opacity 1.5s ease-out;
            }
        `;
        document.head.appendChild(style);

        // Particle Generation for Writer Section
        function createParticles() {
            const writerSection = document.getElementById('writer');
            const particleCount = 20;
            
            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                
                const size = Math.random() * 4 + 2 + 'px';
                particle.style.width = size;
                particle.style.height = size;
                
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                
                const duration = Math.random() * 10 + 10 + 's';
                const delay = Math.random() * 10 + 's';
                particle.style.animation = `float ${duration} linear ${delay} infinite`;
                
                writerSection.appendChild(particle);
            }
        }
